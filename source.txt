class T001_HIPOTECA;
		
	global
		/* No hay declaraciones aquí
		
	create inicio;

	feature inicio is
		local
			i: INTEGER;
		do
			for(i := 0; ; i < 10; i := i + 1;){
				println i;
			}
	end
		
		
end

main
	inicio();
end /* HIPOTECA
